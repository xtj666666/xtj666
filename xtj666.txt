        function showRabbitNBLegend() {

            gloryContainer.innerHTML = '';

            const bigTextDiv = document.createElement('div');

         * 处理跳转: 新标签页打开外部链接 (安全，无拦截)
         * @param {string} url 完整的网址
         */
        function openExternalLink(url) {
            if (!url) return;
            // 确保url有协议头，如果没有则补充https，但传入均已包含协议，安全起见二次确保
            let safeUrl = url;
            if (!safeUrl.startsWith('http://') && !safeUrl.startsWith('https://')) {
                safeUrl = 'https://' + safeUrl;
            }
            // 新窗口打开，并且添加 noopener 增加安全性
            window.open(safeUrl, '_blank', 'noopener,noreferrer');
        }
        
        // 获取所有导航项（标题按钮）
        const navItems = document.querySelectorAll('.nav-item');

            item.addEventListener('click', (event) => {
                // 阻止事件冒泡，避免奇怪影响（无父级干扰）
                event.stopPropagation();
                const type = item.getAttribute('data-type');
                
                if (type === 'url') {
                    // 跳转类型: 获取 data-url 属性并打开新标签页
                    const targetUrl = item.getAttribute('data-url');
                    if (targetUrl) {
                        openExternalLink(targetUrl);
                    } else {
                        console.warn('未找到对应网址');
                    }
                } 
                else if (type === 'special') {
                    // 特殊类型: 第四个标题 “小兔几” => 显示彩色大字 “小兔几NB”
                    showRabbitNBLegend();
                }
                else {
                    // 兜底：啥也不做，理论上只有两种类型
                }
            });
            
            // 增加键盘访问支持，让 accessibility 更好，按Enter同样触发
            item.addEventListener('keydown', (e) => {
                if (e.key === 'Enter' || e.key === ' ') {
                    e.preventDefault();
                    item.click();
                }
            });
            // 设置角色和可聚焦，提高可用性
            item.setAttribute('role', 'button');
            item.setAttribute('tabindex', '0');
        });

            // 同时确保背景样式无闪烁
